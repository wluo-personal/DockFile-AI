sudo docker build -t toneymall/torch2.2-gpu .



RUN docker command
# ==========================
# 1. simple
sudo docker run -it --name tccustom --gpus all toneymall/torch2.2-gpu

# 2. with volume and port mapping
# 6606 is the tensorboard
sudo docker run -it -d -v ~/data:/home/wei/data -p 8891:8888 --ipc=host --name tccustom --rm --gpus all toneymall/torch2.2-gpu

STOP docker command
################################
sudo docker rm -f tccustom



### post build
sudo docker exec -it tccustom bash
conda activate tc-gpu
pip install poetry
# then use poetry to install dependent libs
